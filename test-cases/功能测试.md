## 自动化测试脚本

### Selenium测试脚本示例
```python
# test_registration.py
from selenium import webdriver
from selenium.webdriver.common.by import By
import time

def test_valid_registration():
    driver = webdriver.Chrome()
    try:
        driver.get("http://localhost:8000/register.html")
        
        # 填写注册信息
        driver.find_element(By.ID, "username").send_keys("autotestuser")
        driver.find_element(By.ID, "email").send_keys(f"autotest{int(time.time())}@example.com")
        driver.find_element(By.ID, "password").send_keys("Test123456")
        driver.find_element(By.ID, "confirmPassword").send_keys("Test123456")
        driver.find_element(By.ID, "verificationCode").send_keys("1234")
        driver.find_element(By.ID, "agreeTerms").click()
        
        # 提交注册
        driver.find_element(By.ID, "registerBtn").click()
        time.sleep(2)
        
        # 验证注册成功
        assert "login" in driver.current_url.lower()
        print("✓ 注册测试通过")
        
    finally:
        driver.quit()

if __name__ == "__main__":
    test_valid_registration()
```
